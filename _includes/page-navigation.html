{%- comment -%} A fully automatic page navigation. It sorts all pages in the collection by their file path, which respects the numbered folder structure (e.g., "01-", "02-"). No "order" front matter is needed. {%- endcomment -%} {%- assign section = page.section -%} {%- assign collection = site[section] -%} {%- if collection and collection.size > 1 -%} {# Create a stable, sorted "playlist" of all pages by their file path. #} {%- assign sorted_playlist = collection | sort: "path" -%} {# Find the current page's index in our new playlist. #} {%- for item in sorted_playlist -%} {%- if item.url == page.url -%} {%- assign current_index = forloop.index0 -%} {%- break -%} {%- endif -%} {%- endfor -%} {# Use the index to get the previous and next items directly. #} {%- if current_index != nil -%} {%- if current_index > 0 -%} {%- assign previous_index = current_index | minus: 1 -%} {%- assign previous_page = sorted_playlist[previous_index] -%} {%- endif -%} {%- if current_index < sorted_playlist.size | minus: 1 -%} {%- assign next_index = current_index | plus: 1 -%} {%- assign next_page = sorted_playlist[next_index] -%} {%- endif -%} {%- endif -%}
{%- if previous_page -%} ← {{ previous_page.title }} {%- endif -%} {%- if next_page -%} {{ next_page.title }} → {%- endif -%}
{%- endif -%}
